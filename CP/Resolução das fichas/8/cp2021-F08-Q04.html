Resolução¶

$\def\cata#1{\mathopen{(\!|}#1\mathclose{|\!)}}\def\kons#1{\underline{#1}}$ $unzip \ xs = (map \ \pi_1 \ xs, map \ \pi_2 \ xs)$

{ pointfree; def-split, lei (76) }

$\equiv unzip = <map \ \pi_1 , map \ \pi_2>$

{ def-map-cata, lei (50); $T \ f = map \ f$ }

$\equiv unzip = <\cata{in . B \ (\pi_1,id)},\cata{in . B \ (\pi_2,id)}>$

{ banana-split, lei (53) ; absorção-$\times$ }

$\equiv unzip = \cata{<in . B \ (\pi_1,id) . F \ \pi_1, in . B \ (\pi_2,id) . F \ \pi_2>}$

{ $B(f,id) . F \ f = B (f,id) . B(id,f) = B(f,f)$ }

$\equiv unzip = \cata{<in . B \ (\pi_1, \pi_1), in . B \ (\pi_2, \pi_2)>}$

{ $in = [nil, cons]$; absorçao-+; $B(f,g) = id + f \times g$; def-const ; universal-cata }

$\equiv unzip . in = <[nil,cons.(\pi_1 \times \pi_1)],[nil,cons.(\pi_2 \times \pi_2)]> . (id + id \times unzip)$

{ lei da troca }

$\equiv unzip . in = [<nil, nil>,<cons.(\pi_1 \times \pi_1),cons.(\pi_2 \times \pi_2)>] . (id + id \times unzip)$

{ absorção-+; $in = [nil,cons]$; fusão-+; natural-id }

$\equiv [unzip . nil, unzip . cons] = [<nil, nil> . id,<cons.(\pi_1 \times \pi_1),cons.(\pi_2 \times \pi_2)> . (id \times unzip)]$

{ eq-+; absorção-$\times$ }

$unzip . nil = <nil, nil>$

$unzip . cons = (cons \times cons) . <\pi_1 \times \pi_1,\pi_2 \times \pi_2> . (id \times unzip)$

{ pointwise, def-split, def-$\times$; $nil = \kons{[]}$ }

$unzip \ [] = ([],[])$

$unzip \ (h : t) = (\pi_1 \ h : \pi_1 \ unzip \ t, \pi_2 \ h : \pi_2 \ unzip \ t)$

{ $h = (a,b)$ ; $t = xs$; $(as, bs) = unzip \ xs$ }

$unzip \ [] = ([],[])$

$unzip \ ((a,b) : xs) = (a : as, b : bs) \ where \ (as,bs) = unzip \ xs$
